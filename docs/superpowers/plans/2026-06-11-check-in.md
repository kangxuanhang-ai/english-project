```python
from datetime import datetime, date


async def check_in(db: AsyncSession, user_id: str) -> dict:
    """
    用户每日打卡。
    如果今天已打卡，返回当前 day_number。
    如果今天未打卡，day_number += 1。
    """
    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()
    if not user:
        raise ValueError("用户不存在")

    today = date.today()
    # last_login_at 复用为最后打卡时间
    already_checked = user.last_login_at and user.last_login_at.date() == today

    if already_checked:
        return {"dayNumber": user.day_number, "checkedIn": False}

    user.day_number = (user.day_number or 0) + 1
    user.last_login_at = datetime.now()
    await db.commit()
    await db.refresh(user)
    return {"dayNumber": user.day_number, "checkedIn": True}
```

- [ ] **Step 1: 添加打卡按钮和逻辑**

Read `c:/Users/21943/Desktop/project/group-project/english/apps/web/src/views/Home/index.vue` first.

在 script 部分添加：

```typescript
import { checkIn } from '@/apis/user'
import { useUserStore } from '@/stores/user'
import { ElMessage } from 'element-plus'

const userStore = useUserStore()
const isCheckedIn = computed(() => userStore.isCheckedInToday)
const isAnimating = ref(false)

const handleCheckIn = async () => {
    if (!userStore.getUser) {
        // 未登录，触发登录弹窗（需要父组件 provide IS_SHOW_LOGIN）
        return
    }
    if (isCheckedIn.value) return

    try {
        const res = await checkIn()
        if (res.success) {
            userStore.updateDayNumber(res.data.dayNumber)
            userStore.setCheckedInToday()
            isAnimating.value = true
            setTimeout(() => { isAnimating.value = false }, 1000)
            ElMessage.success(res.message)
        }
    } catch (e) {
        // 错误已由 aiApi 拦截器处理
    }
}
```

在模板的"坚持5天打卡学习"区域下方添加打卡按钮：

```html
<div class="flex items-center gap-4 pt-4">
    <button
        v-if="!userStore.getUser"
        @click="handleCheckIn"
        class="px-6 py-2 bg-indigo-500 text-white rounded-full text-sm hover:bg-indigo-600 transition-colors">
        登录后打卡
    </button>
    <button
        v-else-if="!isCheckedIn"
        @click="handleCheckIn"
        class="px-6 py-2 bg-indigo-500 text-white rounded-full text-sm hover:bg-indigo-600 transition-colors">
        📌 打卡
    </button>
    <span
        v-else
        class="px-6 py-2 bg-gray-200 text-gray-500 rounded-full text-sm cursor-not-allowed">
        ✅ 已打卡
    </span>
    <span
        :class="{ 'animate-bounce text-indigo-400': isAnimating }"
        class="text-white text-sm">
        累计 {{ userStore.getUser?.dayNumber ?? 0 }} 天
    </span>
</div>
```
